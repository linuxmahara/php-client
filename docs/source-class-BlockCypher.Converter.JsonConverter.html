 74      * @param mixed $value
 75      * @param int $options
 76      * @param int $depth
 77      * @return string
 78      */
 79     public static function encode($value, $options = 0, $depth = 512)
 80     {
 81         // 5.6.6    JSON_PRESERVE_ZERO_FRACTION option was added.
 82         // 5.5.0    depth parameter was added.
 83         // 5.4.0    JSON_PRETTY_PRINT, JSON_UNESCAPED_SLASHES, and JSON_UNESCAPED_UNICODE options were added.
 84         // 5.3.3    JSON_NUMERIC_CHECK option was added.
 85         // 5.3.0    The options parameter was added.
 86 
 87         if (version_compare(phpversion(), '5.5.0', '>=') === true) {
 88             return json_encode($value, $options, $depth);
 89         }
 90 
 91         if (version_compare(phpversion(), '5.4.0', '>=') === true) {
 92             return json_encode($value, $options);
 93         }
 94 
 95         if (version_compare(phpversion(), '5.3.3', '>=') === true) {
 96             $json = json_encode($value, $options);
 97             // Available since PHP 5.4.0.
 98             if ($options & JSON_UNESCAPED_SLASHES) {
 99                 $json = str_replace('\\/', '/', $json);
100             }
101             return $json;
102         }
103 
104         if (version_compare(phpversion(), '5.3.0', '>=') === true) {
105             $json = json_encode($value, $options);
106             // Available since PHP 5.3.3.
107             if ($options & JSON_NUMERIC_CHECK) {
108                 // TODO: implement option for lower PHP versions
109                 // IF string contains number then remove "
110             }
111             return $json;
112         }
113 
114         $json = json_encode($value, $options);
115 
116         // Available since PHP 5.3.0.
117         if ($options & JSON_HEX_AMP) {
118             $json = str_replace('&', "\\u0026", $json);
119         }
120 
121         return $json;
122     }

137         // 5.2.1    Added support for JSON decoding of basic types.
138 
139         if (version_compare(phpversion(), '5.4.0', '>=') === true) {
140             return json_decode($json, $assoc, $depth, $options);
141         }
142 
143         // Available since PHP 5.4.0.
144         if ($options & JSON_BIGINT_AS_STRING) {
145             $json = preg_replace('/\: *([0-9]+\.?[0-9e+\-]*)/', ':"\\1"', $json);
146         }
147 
148         if (version_compare(phpversion(), '5.3.0', '>=') === true) {
149             return json_decode($json, $assoc, $depth);
150         }
151
