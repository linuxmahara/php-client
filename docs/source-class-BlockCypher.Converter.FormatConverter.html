37         if ($currency && array_key_exists($currency, $currencyDecimals)) {
38             if (strpos($value, ".") !== false && (floor($value) != $value)) {
39                 //throw exception if it has decimal values for JPY, TWD and other virtual currencies which do not end with .00
40                 throw new \InvalidArgumentException("value cannot have decimals for $currency currency");
41             }
42             $decimals = $currencyDecimals[$currency];
43         } else if (strpos($value, ".") === false) {
44             // Check if value has decimal values. If not no need to assign 2 decimals with .00 at the end
45             $decimals = 0;
46         }
